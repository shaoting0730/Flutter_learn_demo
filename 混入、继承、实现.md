// 抽象类：有类继承自抽象类时,需要重写其抽象方法
// 混入类：混入类不能有构造方法，混入类不能被继承
// 实现implements： 需要实现父类的所有属性和方法（包括抽象方法和非抽象方法）



// 普通类
class A {
  var a = 0;
  printA() {
    print('a');
  }
}

// 混入类 
mixin B {
  var b = 0;
  // B(this.b);  // 混入类不能有构造方法  
  printB() {
    print('b');
  }
}

// 抽象类   有类继承自抽象类时,需要重写其抽象方法 
abstract class C {
  var c = 0;
  printC();    /// 抽象方法  抽象方法一定得定义在抽象类

  printCC() {
    print('我是抽象类中的非抽象方法');
  }
}

/// 继承
class One extends A {
  onePrint() {
    print(super.a);
    print(super.printA());
  }
}

// class ONe extends B{  // 混入类不能被继承
//   // error:
//   // Classes can only extend other classes.
//   // Try specifying a different superclass, or removing the extends clause.dartextends_non_class
// }
class ONE extends C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  onePrint() {
    print(super.c);
    print(super.printCC());
  }
}

/// 实现   需要实现父类的所有属性和方法（包括抽象方法和非抽象方法）
class Two implements A {
  @override
  late int a;
  @override
  printA() {
    // TODO: implement printA
    throw UnimplementedError();
  }
}

class TWo implements B {
  @override
  late int b;
  @override
  printB() {
    // TODO: implement printB
    throw UnimplementedError();
  }
}

class TWO implements C {
  @override
  late int c;

  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  @override
  printCC() {
    // TODO: implement printCC
    throw UnimplementedError();
  }
}

/// with 混入 类似于多继承
class Three with A {
  printThree() {
    print(super.a);
    print(super.printA());
  }
}

class THree with B {
  printThree() {
    print(super.b);
    print(super.printB());
  }
}

class THRee with C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  printThree() {
    print(super.c);
    print(super.printCC());
    // print(super.printC());  // error: The method 'printC' is always abstract in the supertype.dartabstract_super_member_reference
  }
}

class THREe with A, B, C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  printThree() {
    print(super.a);
    print(super.printA());

    print(super.b);
    print(super.printB());

    print(super.c);
    print(super.printCC());
    // print(super.printC());  // error: The method 'printC' is always abstract in the supertype.dartabstract_super_member_reference
  }
}
